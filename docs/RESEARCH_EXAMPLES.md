```python
# Custom analysis script for detailed compiler comparison
import json
from pathlib import Path

def analyze_compiler_patterns(results_dir):
    # Load compiler fingerprinting results
    with open(results_dir / "compiler_fingerprinting.json") as f:
        data = json.load(f)
    
    # Extract per-binary compiler predictions
    compiler_dist = data['corpus_summary']['compiler_distribution']
    per_binary = data['per_binary_analysis']
    
    # Analyze feature differences by predicted compiler
    gcc_features = []
    clang_features = []
    rustc_features = []
    
    for binary_name, analysis in per_binary.items():
        predicted = analysis['predicted_compiler']
        features = analysis['features']
        
        if predicted == 'gcc':
            gcc_features.append(features)
        elif predicted == 'clang':  
            clang_features.append(features)
        elif predicted == 'rustc':
            rustc_features.append(features)
    
    return {
        'distribution': compiler_dist,
        'gcc_patterns': analyze_feature_means(gcc_features),
        'clang_patterns': analyze_feature_means(clang_features),
        'rustc_patterns': analyze_feature_means(rustc_features)
    }

def analyze_feature_means(feature_list):
    if not feature_list:
        return {}
        
    import numpy as np
    feature_means = {}
    
    # Get all feature names
    all_features = set()
    for features in feature_list:
        all_features.update(features.keys())
    
    # Calculate means and standard deviations
    for feature in all_features:
        values = [features.get(feature, 0) for features in feature_list]
        feature_means[feature] = {
            'mean': np.mean(values),
            'std': np.std(values),
            'count': len(values)
        }
    
    return feature_means

# Run analysis
results = analyze_compiler_patterns(Path('./compiler_results'))
print(f"Compiler Distribution: {results['distribution']}")
```

def discover_universal_patterns(results_dir):
    """Mine for universal instruction patterns across corpus."""
    import json
    from collections import Counter, defaultdict
    
    # Load analysis results
    with open(results_dir / "fast_analysis_raw.json") as f:
        data = json.load(f)
    
    # Aggregate patterns across all binaries
    all_instructions = []
    all_bigrams = []
    all_trigrams = []
    
    for binary_result in data['per_binary_results']:
        instructions = binary_result.get('sampled_sequence', [])
        all_instructions.extend(instructions)
        
        # Extract n-grams
        bigrams = [(instructions[i], instructions[i+1]) 
                  for i in range(len(instructions)-1)]
        trigrams = [(instructions[i], instructions[i+1], instructions[i+2])
                   for i in range(len(instructions)-2)]
        
        all_bigrams.extend(bigrams)
        all_trigrams.extend(trigrams)
    
    # Find universal patterns (appear in >80% of binaries)
    total_binaries = len(data['per_binary_results'])
    min_binary_count = int(0.8 * total_binaries)
    
    # Count pattern frequencies
    instruction_counter = Counter(all_instructions)
    bigram_counter = Counter([' '.join(bg) for bg in all_bigrams])
    trigram_counter = Counter([' '.join(tg) for tg in all_trigrams])
    
    # Identify universal patterns
    universal_instructions = [
        (instr, count) for instr, count in instruction_counter.most_common(20)
        if count > min_binary_count
    ]
    
    universal_bigrams = [
        (pattern, count) for pattern, count in bigram_counter.most_common(50)
        if count > min_binary_count // 10  # More lenient for sequences
    ]
    
    return {
        'corpus_size': total_binaries,
        'total_instructions': len(all_instructions),
        'universal_instructions': universal_instructions,
        'universal_bigrams': universal_bigrams[:20],  # Top 20
        'instruction_diversity': len(instruction_counter),
        'pattern_diversity': len(bigram_counter)
    }

# Run pattern discovery
patterns = discover_universal_patterns(Path('./pattern_discovery'))
```

    trends = {}

    instruction_trends = {}
    periods = temporal_results['periods']
    
    for instruction in ['mov', 'lea', 'vector_*', 'cmov*']:
        usage_over_time = []
        for period in periods:

def compare_languages(analysis_results):
    """Compare compilation patterns across programming languages."""
    
    # Load compiler fingerprinting results with language metadata
    language_patterns = defaultdict(lambda: {
        'binaries': [],
        'features': [],
        'optimizations': [],
        'instruction_stats': []
    })
    
    per_binary = analysis_results['per_binary_analysis']
    
    for binary_name, analysis in per_binary.items():
        # Extract language from metadata (if available)
        lang = analysis.get('metadata_info', {}).get('source_language', 'unknown')
        
        if lang != 'unknown':
            language_patterns[lang]['binaries'].append(binary_name)
            language_patterns[lang]['features'].append(analysis['features'])
            language_patterns[lang]['optimizations'].append(analysis['predicted_optimization'])
    
    # Compute language-specific statistics
    language_comparison = {}
    
    for lang, data in language_patterns.items():
        if len(data['binaries']) >= 5:  # Minimum sample size
            feature_means = compute_feature_means(data['features'])
            opt_distribution = Counter(data['optimizations'])
            
            language_comparison[lang] = {
                'sample_size': len(data['binaries']),
                'feature_signature': feature_means,
                'optimization_preference': dict(opt_distribution),
                'distinctive_features': identify_distinctive_features(lang, feature_means, language_patterns)
            }
    
    return language_comparison

def identify_distinctive_features(target_lang, target_features, all_language_patterns):
    """Identify features that distinguish this language from others."""
    distinctive = {}
    
    for feature, stats in target_features.items():
        target_mean = stats['mean']
        
        # Compare with other languages
        other_means = []
        for lang, data in all_language_patterns.items():
            if lang != target_lang and len(data['features']) >= 3:
                lang_features = compute_feature_means(data['features'])
                if feature in lang_features:
                    other_means.append(lang_features[feature]['mean'])
        
        if other_means:
            other_mean = np.mean(other_means)
            difference_ratio = target_mean / max(other_mean, 1e-6)
            
            # Feature is distinctive if >2x different from average
            if difference_ratio > 2.0 or difference_ratio < 0.5:
                distinctive[feature] = {
                    'target_mean': target_mean,
                    'others_mean': other_mean,
                    'ratio': difference_ratio
                }
    
    return distinctive
```

## Statistical Significance and Validation

### Sample Size Calculations
For statistically significant results:
- **Compiler Comparison**: n≥20 per compiler for 95% confidence
- **Optimization Detection**: n≥30 per optimization level  
- **Language Comparison**: n≥25 per language for robust statistics
- **Pattern Discovery**: n≥100 total binaries for universal patterns

### Confidence Intervals
All statistical measures include:
```python
import scipy.stats as stats

def compute_confidence_interval(data, confidence=0.95):
    """Compute confidence interval for statistical measures."""
    n = len(data)
    mean = np.mean(data)
    sem = stats.sem(data)  # Standard error of mean
    h = sem * stats.t.ppf((1 + confidence) / 2, n-1)
    
    return {
        'mean': mean,
        'confidence_interval': (mean - h, mean + h),
        'margin_of_error': h
    }
```

### Validation Protocols
1. **Cross-Validation**: k-fold validation for all classifiers
2. **Hold-Out Testing**: 20% of data reserved for final validation
3. **Bootstrap Resampling**: Confidence intervals for all statistics
4. **Reproducibility**: Fixed random seeds, documented parameters
